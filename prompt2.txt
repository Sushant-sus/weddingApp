## STORED PROCEDURES — 006_sp_auth.sql

Generate ALL of the following stored procedures:

### User Registration
CREATE OR REPLACE FUNCTION wedding.sp_auth_register(
  p_full_name TEXT,
  p_email     TEXT,
  p_password_hash TEXT
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE
  v_user_id UUID;
  v_viewer_role_id UUID;
BEGIN
  -- Check if email already exists
  IF EXISTS (SELECT 1 FROM wedding.users WHERE email = p_email AND deleted_at IS NULL) THEN
    RAISE EXCEPTION 'EMAIL_ALREADY_EXISTS: Email % is already registered', p_email;
  END IF;

  -- Get default VIEWER role
  SELECT id INTO v_viewer_role_id FROM wedding.roles WHERE name = 'VIEWER';

  INSERT INTO wedding.users(full_name, email, password_hash, role_id)
  VALUES (p_full_name, p_email, p_password_hash, v_viewer_role_id)
  RETURNING id INTO v_user_id;

  RETURN jsonb_build_object('user_id', v_user_id, 'email', p_email);
EXCEPTION WHEN OTHERS THEN RAISE;
END; $$;

### Get User For Login (returns password_hash for comparison)
CREATE OR REPLACE FUNCTION wedding.sp_auth_get_user_by_email(p_email TEXT)
RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_result JSONB;
BEGIN
  SELECT row_to_json(r) INTO v_result FROM (
    SELECT
      u.id, u.full_name, u.email, u.password_hash,
      u.is_active, u.is_email_verified,
      r.name AS role_name, r.id AS role_id,
      COALESCE(
        json_agg(p.name) FILTER (WHERE p.name IS NOT NULL), '[]'
      ) AS permissions
    FROM wedding.users u
    LEFT JOIN wedding.roles r ON r.id = u.role_id
    LEFT JOIN wedding.role_permissions rp ON rp.role_id = r.id
    LEFT JOIN wedding.permissions p ON p.id = rp.permission_id
    WHERE u.email = p_email AND u.deleted_at IS NULL
    GROUP BY u.id, r.name, r.id
  ) r;

  IF v_result IS NULL THEN
    RAISE EXCEPTION 'USER_NOT_FOUND: No user found with email %', p_email;
  END IF;

  RETURN v_result;
END; $$;

### Save Refresh Token
CREATE OR REPLACE FUNCTION wedding.sp_auth_save_refresh_token(
  p_user_id    UUID,
  p_token_hash TEXT,
  p_expires_at TIMESTAMPTZ
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_id UUID;
BEGIN
  -- Revoke all existing refresh tokens for user (single session)
  UPDATE wedding.refresh_tokens
  SET revoked_at = NOW()
  WHERE user_id = p_user_id AND revoked_at IS NULL;

  INSERT INTO wedding.refresh_tokens(user_id, token_hash, expires_at)
  VALUES (p_user_id, p_token_hash, p_expires_at)
  RETURNING id INTO v_id;

  RETURN jsonb_build_object('token_id', v_id);
END; $$;

### Validate Refresh Token
CREATE OR REPLACE FUNCTION wedding.sp_auth_validate_refresh_token(p_token_hash TEXT)
RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_result JSONB;
BEGIN
  SELECT row_to_json(r) INTO v_result FROM (
    SELECT rt.user_id, rt.expires_at,
           u.email, u.full_name, u.is_active,
           ro.name AS role_name,
           COALESCE(
             json_agg(p.name) FILTER (WHERE p.name IS NOT NULL), '[]'
           ) AS permissions
    FROM wedding.refresh_tokens rt
    JOIN wedding.users u ON u.id = rt.user_id
    LEFT JOIN wedding.roles ro ON ro.id = u.role_id
    LEFT JOIN wedding.role_permissions rp ON rp.role_id = ro.id
    LEFT JOIN wedding.permissions p ON p.id = rp.permission_id
    WHERE rt.token_hash = p_token_hash
      AND rt.revoked_at IS NULL
      AND rt.expires_at > NOW()
      AND u.deleted_at IS NULL
    GROUP BY rt.user_id, rt.expires_at, u.email, u.full_name, u.is_active, ro.name
  ) r;

  IF v_result IS NULL THEN
    RAISE EXCEPTION 'INVALID_REFRESH_TOKEN: Token is invalid or expired';
  END IF;

  RETURN v_result;
END; $$;

### Revoke Refresh Token (logout)
CREATE OR REPLACE FUNCTION wedding.sp_auth_revoke_refresh_token(p_user_id UUID)
RETURNS JSONB LANGUAGE plpgsql AS $$
BEGIN
  UPDATE wedding.refresh_tokens
  SET revoked_at = NOW()
  WHERE user_id = p_user_id AND revoked_at IS NULL;
  RETURN jsonb_build_object('logged_out', TRUE);
END; $$;

### Save OTP
CREATE OR REPLACE FUNCTION wedding.sp_auth_save_otp(
  p_email     TEXT,
  p_code_hash TEXT,
  p_type      TEXT
) RETURNS JSONB LANGUAGE plpgsql AS $$
BEGIN
  -- Invalidate previous OTPs of same type for this email
  UPDATE wedding.otp_codes
  SET used_at = NOW()
  WHERE email = p_email AND type = p_type AND used_at IS NULL;

  INSERT INTO wedding.otp_codes(email, code_hash, type, expires_at)
  VALUES (p_email, p_code_hash, p_type, NOW() + INTERVAL '10 minutes');

  RETURN jsonb_build_object('otp_sent', TRUE, 'expires_in', '10 minutes');
END; $$;

### Verify OTP
CREATE OR REPLACE FUNCTION wedding.sp_auth_verify_otp(
  p_email     TEXT,
  p_code_hash TEXT,
  p_type      TEXT
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_otp_id UUID;
BEGIN
  SELECT id INTO v_otp_id
  FROM wedding.otp_codes
  WHERE email      = p_email
    AND code_hash  = p_code_hash
    AND type       = p_type
    AND used_at    IS NULL
    AND expires_at > NOW()
  ORDER BY created_at DESC
  LIMIT 1;

  IF v_otp_id IS NULL THEN
    RAISE EXCEPTION 'INVALID_OTP: OTP is invalid or has expired';
  END IF;

  -- Mark OTP as used
  UPDATE wedding.otp_codes SET used_at = NOW() WHERE id = v_otp_id;

  -- If email verification OTP, mark user email as verified
  IF p_type = 'VERIFY_EMAIL' THEN
    UPDATE wedding.users
    SET is_email_verified = TRUE, updated_at = NOW()
    WHERE email = p_email;
  END IF;

  RETURN jsonb_build_object('verified', TRUE);
END; $$;

### Update Last Login
CREATE OR REPLACE FUNCTION wedding.sp_auth_update_last_login(p_user_id UUID)
RETURNS VOID LANGUAGE plpgsql AS $$
BEGIN
  UPDATE wedding.users SET last_login_at = NOW() WHERE id = p_user_id;
END; $$;

### Admin — Get All Users
CREATE OR REPLACE FUNCTION wedding.sp_admin_get_users(
  p_role_name TEXT DEFAULT NULL,
  p_is_active BOOLEAN DEFAULT NULL
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_result JSONB;
BEGIN
  SELECT json_agg(row_to_json(r)) INTO v_result FROM (
    SELECT
      u.id, u.full_name, u.email, u.is_active,
      u.is_email_verified, u.last_login_at, u.created_at,
      r.name AS role_name, r.id AS role_id
    FROM wedding.users u
    LEFT JOIN wedding.roles r ON r.id = u.role_id
    WHERE u.deleted_at IS NULL
      AND (p_role_name IS NULL OR r.name = p_role_name)
      AND (p_is_active IS NULL OR u.is_active = p_is_active)
    ORDER BY u.created_at DESC
  ) r;
  RETURN COALESCE(v_result, '[]'::JSONB);
END; $$;

### Admin — Assign Role to User
CREATE OR REPLACE FUNCTION wedding.sp_admin_assign_role(
  p_user_id UUID,
  p_role_id UUID
) RETURNS JSONB LANGUAGE plpgsql AS $$
BEGIN
  UPDATE wedding.users
  SET role_id = p_role_id, updated_at = NOW()
  WHERE id = p_user_id AND deleted_at IS NULL;

  IF NOT FOUND THEN
    RAISE EXCEPTION 'USER_NOT_FOUND';
  END IF;

  RETURN jsonb_build_object('role_assigned', TRUE, 'user_id', p_user_id);
END; $$;

### Admin — Toggle User Active Status
CREATE OR REPLACE FUNCTION wedding.sp_admin_toggle_user_status(p_user_id UUID)
RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_new_status BOOLEAN;
BEGIN
  UPDATE wedding.users
  SET is_active = NOT is_active, updated_at = NOW()
  WHERE id = p_user_id AND deleted_at IS NULL
  RETURNING is_active INTO v_new_status;

  IF NOT FOUND THEN RAISE EXCEPTION 'USER_NOT_FOUND'; END IF;

  RETURN jsonb_build_object('is_active', v_new_status, 'user_id', p_user_id);
END; $$;

### Get All Roles (for dropdown in admin panel)
CREATE OR REPLACE FUNCTION wedding.sp_admin_get_roles()
RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_result JSONB;
BEGIN
  SELECT json_agg(row_to_json(r)) INTO v_result FROM (
    SELECT ro.id, ro.name, ro.description,
           COALESCE(json_agg(p.name) FILTER (WHERE p.name IS NOT NULL), '[]') AS permissions
    FROM wedding.roles ro
    LEFT JOIN wedding.role_permissions rp ON rp.role_id = ro.id
    LEFT JOIN wedding.permissions p ON p.id = rp.permission_id
    GROUP BY ro.id, ro.name, ro.description
    ORDER BY ro.name
  ) r;
  RETURN COALESCE(v_result, '[]'::JSONB);
END; $$;

## AUTH SERVICE LOGIC (auth.service.ts)

### register(data):
1. Hash password with bcrypt (rounds: 12)
2. Call wedding.sp_auth_register(fullName, email, passwordHash)
3. Generate 6-digit OTP using crypto.randomInt(100000, 999999)
4. Hash OTP with bcrypt
5. Call wedding.sp_auth_save_otp(email, otpHash, 'VERIFY_EMAIL')
6. Send OTP email via nodemailer
7. Return { message: 'OTP sent to your email. Please verify.' }

### verifyEmail(email, otp):
1. Hash incoming OTP
2. Call wedding.sp_auth_verify_otp(email, hashedOtp, 'VERIFY_EMAIL')
3. Return { message: 'Email verified successfully. You can now login.' }

### login(email, password):
1. Call wedding.sp_auth_get_user_by_email(email)
2. Check is_active — if false throw USER_INACTIVE
3. Check is_email_verified — if false throw EMAIL_NOT_VERIFIED
4. bcrypt.compare(password, passwordHash)
5. If mismatch throw INVALID_CREDENTIALS
6. Generate accessToken (JWT, expires: 15m) with payload:
   { userId, email, role, permissions }
7. Generate refreshToken (JWT, expires: 7d)
8. Hash refreshToken and call sp_auth_save_refresh_token
9. Call sp_auth_update_last_login
10. Return { accessToken, refreshToken, user: { id, fullName, email, role, permissions } }

### refreshToken(token):
1. Verify JWT signature of incoming refresh token
2. Hash it and call sp_auth_validate_refresh_token
3. Issue new accessToken
4. Return { accessToken }

### logout(userId):
1. Call sp_auth_revoke_refresh_token(userId)
2. Return { message: 'Logged out successfully' }

### forgotPassword(email):
1. Check user exists via sp_auth_get_user_by_email
2. Generate OTP, hash it
3. Call sp_auth_save_otp(email, hash, 'RESET_PASSWORD')
4. Send OTP via email
5. Return { message: 'Password reset OTP sent to your email' }

### resetPassword(email, otp, newPassword):
1. Hash OTP and call sp_auth_verify_otp(email, hash, 'RESET_PASSWORD')
2. Hash new password with bcrypt
3. Update user password_hash directly
4. Revoke all refresh tokens for user
5. Return { message: 'Password reset successfully' }

## AUTH MIDDLEWARE (auth.middleware.ts)

### authenticate (JWT guard):
- Extract Bearer token from Authorization header
- Verify JWT with ACCESS_TOKEN_SECRET
- Attach decoded user to req.user
- Throw 401 if missing or invalid

### requireRole(...roles):
- Check req.user.role is in allowed roles array
- Throw 403 if not

### requirePermission(...permissions):
- Check req.user.permissions includes ALL required permissions
- Throw 403 if not

## FRONTEND AUTH FLOW

### Register Flow:
1. User fills RegisterForm (full name, email, password, confirm password)
2. POST /auth/register → success
3. Redirect to VerifyEmailPage
4. User enters 6-digit OTP received in email
5. POST /auth/verify-email → success
6. Redirect to LoginPage with success toast

### Login Flow:
1. User fills LoginForm (email, password)
2. POST /auth/login → receives { accessToken, refreshToken, user }
3. Store accessToken in memory (React state/context)
4. Store refreshToken in httpOnly cookie (or localStorage as fallback)
5. Store user info (name, role, permissions) in AuthContext
6. Redirect to Dashboard

### Token Refresh Flow (axios interceptor):
1. Every API request attaches Authorization: Bearer <accessToken>
2. If 401 received → interceptor calls POST /auth/refresh-token
3. Gets new accessToken → retries original request
4. If refresh also fails → logout user → redirect to login

### Logout Flow:
1. POST /auth/logout
2. Clear AuthContext state
3. Clear stored tokens
4. Redirect to LoginPage

### Protected Routes:
<ProtectedRoute allowedRoles={['SUPERADMIN', 'ADMIN']}>
  <AdminUsersPage />
</ProtectedRoute>

<ProtectedRoute allowedPermissions={['guests:write']}>
  <GuestManagementPage />
</ProtectedRoute>

## ZOD VALIDATION SCHEMAS

### Register:
{
  fullName: z.string().min(2).max(100),
  email: z.string().email(),
  password: z.string()
    .min(8)
    .regex(/[A-Z]/, 'Must have uppercase')
    .regex(/[0-9]/, 'Must have number')
    .regex(/[^a-zA-Z0-9]/, 'Must have special character'),
  confirmPassword: z.string()
}.refine(data => data.password === data.confirmPassword, {
  message: 'Passwords do not match',
  path: ['confirmPassword']
})

### Login:
{
  email: z.string().email(),
  password: z.string().min(1)
}

### Verify Email:
{
  email: z.string().email(),
  otp: z.string().length(6).regex(/^\d+$/)
}

### Reset Password:
{
  email: z.string().email(),
  otp: z.string().length(6),
  newPassword: z.string().min(8)... (same rules as register)
}

## CONCEPT

Think of it like Google Docs sharing:
- A user creates a Wedding Event (becomes its OWNER)
- They can invite other users by email
- Each invited user gets an EVENT-SCOPED ROLE for that specific event
- The same user can be LEADER in Wedding A but VIEWER in Wedding B
- Global roles (SUPERADMIN, ADMIN) from the auth system still apply
  on top of event-level roles

---

## EVENT ROLES (Scoped Per Wedding Event)

Define these 5 event-level roles in order of access level:

### OWNER
- Created the wedding event
- Full control: edit event, manage members, delete event
- Can assign/change roles of all other members
- Cannot be removed by anyone except SUPERADMIN
- Only one OWNER per event (transferable)

### LEADER
- Co-organizer level access
- Full CRUD on guests, gifts, itinerary, costs
- Can invite new members and assign roles (except OWNER)
- Cannot delete the wedding event itself

### EDITOR
- Working member access
- Full CRUD on guests, gifts, itinerary, costs
- Cannot manage members or invite others
- Cannot delete the wedding event

### CONTRIBUTOR
- Limited write access
- Can add guests and record gifts received
- Cannot edit or delete existing records
- Cannot touch itinerary or costs

### VIEWER
- Read-only access
- Can view all data but cannot modify anything
- Cannot see cost details (financial privacy)
- Useful for family members who want to track attendance

### Get Event by ID (with caller's role)
CREATE OR REPLACE FUNCTION wedding.sp_event_get_by_id(
  p_event_id UUID,
  p_user_id  UUID
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_result JSONB;
BEGIN
  SELECT row_to_json(r) INTO v_result FROM (
    SELECT
      e.*,
      em.event_role AS my_role,
      (SELECT COUNT(*) FROM wedding.event_members
       WHERE event_id = e.id AND invite_status = 'ACCEPTED') AS member_count
    FROM wedding.wedding_events e
    JOIN wedding.event_members em ON em.event_id = e.id
    WHERE e.id = p_event_id
      AND em.user_id = p_user_id
      AND e.deleted_at IS NULL
  ) r;

  IF v_result IS NULL THEN
    RAISE EXCEPTION 'EVENT_NOT_FOUND_OR_NO_ACCESS';
  END IF;

  RETURN v_result;
END; $$;

### Get All Events for a User
CREATE OR REPLACE FUNCTION wedding.sp_event_get_all_for_user(p_user_id UUID)
RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_result JSONB;
BEGIN
  SELECT json_agg(row_to_json(r)) INTO v_result FROM (
    SELECT
      e.id, e.name, e.wedding_date, e.venue, e.description,
      e.is_active, e.created_at,

### Get All Members of an Event
CREATE OR REPLACE FUNCTION wedding.sp_event_get_members(p_event_id UUID)
RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_result JSONB;
BEGIN
  SELECT json_agg(row_to_json(r)) INTO v_result FROM (
    SELECT
      em.id, em.event_role, em.invite_status, em.joined_at,
      u.id AS user_id, u.full_name, u.email,
      inviter.full_name AS invited_by_name
    FROM wedding.event_members em
    JOIN wedding.users u ON u.id = em.user_id
    LEFT JOIN wedding.users inviter ON inviter.id = em.invited_by
    WHERE em.event_id = p_event_id
    ORDER BY
      CASE em.event_role
        WHEN 'OWNER'       THEN 1
        WHEN 'LEADER'      THEN 2
        WHEN 'EDITOR'      THEN 3
        WHEN 'CONTRIBUTOR' THEN 4
        WHEN 'VIEWER'      THEN 5
      END
  ) r;
  RETURN COALESCE(v_result, '[]'::JSONB);
END; $$;

### Invite Member to Event (by email)
CREATE OR REPLACE FUNCTION wedding.sp_event_invite_member(
  p_event_id   UUID,
  p_inviter_id UUID,
  p_email      TEXT,
  p_event_role TEXT
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE
  v_invitee_id    UUID;
  v_invite_token  TEXT;
  v_member_role   TEXT;
BEGIN
  -- Check inviter has permission (must be OWNER or LEADER)
  SELECT event_role INTO v_member_role
  FROM wedding.event_members
  WHERE event_id = p_event_id AND user_id = p_inviter_id
    AND invite_status = 'ACCEPTED';

  IF v_member_role NOT IN ('OWNER', 'LEADER') THEN
    RAISE EXCEPTION 'INSUFFICIENT_PERMISSION: Only OWNER or LEADER can invite members';
  END IF;

  -- LEADER cannot assign OWNER role
  IF v_member_role = 'LEADER' AND p_event_role = 'OWNER' THEN
    RAISE EXCEPTION 'INSUFFICIENT_PERMISSION: LEADER cannot assign OWNER role';
  END IF;

  -- Find user by email (they must already be registered)
  SELECT id INTO v_invitee_id FROM wedding.users
  WHERE email = p_email AND deleted_at IS NULL AND is_active = TRUE;

  IF v_invitee_id IS NULL THEN
    RAISE EXCEPTION 'USER_NOT_FOUND: No registered user found with email %', p_email;
  END IF;

  -- Check not already a member
  IF EXISTS (
    SELECT 1 FROM wedding.event_members
    WHERE event_id = p_event_id AND user_id = v_invitee_id
  ) THEN
    RAISE EXCEPTION 'ALREADY_A_MEMBER: User is already a member of this event';
  END IF;

  -- Generate secure invite token
  v_invite_token := encode(gen_random_bytes(32), 'hex');

  INSERT INTO wedding.event_members(
    event_id, user_id, event_role, invited_by,
    invite_token, invite_expires_at
  )
  VALUES (
    p_event_id, v_invitee_id, p_event_role, p_inviter_id,
    v_invite_token, NOW() + INTERVAL '48 hours'
  );

  RETURN jsonb_build_object(
    'invite_token', v_invite_token,
    'invitee_email', p_email,
    'event_role', p_event_role,
    'expires_in', '48 hours'
  );
EXCEPTION WHEN OTHERS THEN RAISE;
END; $$;

### Accept Invite
CREATE OR REPLACE FUNCTION wedding.sp_event_accept_invite(
  p_user_id UUID,
  p_invite_token TEXT
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_event_id UUID;
BEGIN
  UPDATE wedding.event_members
  SET invite_status = 'ACCEPTED', joined_at = NOW(),
      invite_token = NULL, updated_at = NOW()
  WHERE user_id = p_user_id
    AND invite_token = p_invite_token
    AND invite_status = 'PENDING'
    AND invite_expires_at > NOW()
  RETURNING event_id INTO v_event_id;

  IF v_event_id IS NULL THEN
    RAISE EXCEPTION 'INVALID_INVITE: Invite token is invalid or expired';
  END IF;

  RETURN wedding.sp_event_get_by_id(v_event_id, p_user_id);
END; $$;

### Change Member Role (OWNER only can promote to LEADER, anyone below by LEADER+)
CREATE OR REPLACE FUNCTION wedding.sp_event_change_member_role(
  p_event_id    UUID,
  p_changer_id  UUID,
  p_target_user_id UUID,
  p_new_role    TEXT
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_changer_role TEXT; v_target_role TEXT;
BEGIN
  SELECT event_role INTO v_changer_role FROM wedding.event_members
  WHERE event_id = p_event_id AND user_id = p_changer_id AND invite_status = 'ACCEPTED';

  SELECT event_role INTO v_target_role FROM wedding.event_members
  WHERE event_id = p_event_id AND user_id = p_target_user_id AND invite_status = 'ACCEPTED';

  -- Cannot change OWNER role
  IF v_target_role = 'OWNER' THEN
    RAISE EXCEPTION 'CANNOT_CHANGE_OWNER_ROLE';
  END IF;

  -- Only OWNER can assign LEADER
  IF p_new_role = 'OWNER' OR (p_new_role = 'LEADER' AND v_changer_role != 'OWNER') THEN
    RAISE EXCEPTION 'INSUFFICIENT_PERMISSION';
  END IF;

  -- Must be OWNER or LEADER to change roles
  IF v_changer_role NOT IN ('OWNER', 'LEADER') THEN
    RAISE EXCEPTION 'INSUFFICIENT_PERMISSION';
  END IF;

  UPDATE wedding.event_members
  SET event_role = p_new_role, updated_at = NOW()
  WHERE event_id = p_event_id AND user_id = p_target_user_id;

  RETURN jsonb_build_object('role_changed', TRUE, 'new_role', p_new_role);
END; $$;

### Remove Member from Event
CREATE OR REPLACE FUNCTION wedding.sp_event_remove_member(
  p_event_id    UUID,
  p_remover_id  UUID,
  p_target_user_id UUID
) RETURNS JSONB LANGUAGE plpgsql AS $$
DECLARE v_remover_role TEXT; v_target_role TEXT;
BEGIN
  SELECT event_role INTO v_remover_role FROM wedding.event_members
  WHERE event_id = p_event_id AND user_id = p_remover_id AND invite_status = 'ACCEPTED';

  SELECT event_role INTO v_target_role FROM wedding.event_members
  WHERE event_id = p_event_id AND user_id = p_target_user_id;

  IF v_target_role = 'OWNER' THEN
    RAISE EXCEPTION 'CANNOT_REMOVE_OWNER';
  END IF;

  IF v_remover_role NOT IN ('OWNER', 'LEADER') THEN
    RAISE EXCEPTION 'INSUFFICIENT_PERMISSION';
  END IF;

  DELETE FROM wedding.event_members
  WHERE event_id = p_event_id AND user_id = p_target_user_id;

  RETURN jsonb_build_object('removed', TRUE, 'user_id', p_target_user_id);
END; $$;

### Transfer Ownership
CREATE OR REPLACE FUNCTION wedding.sp_event_transfer_ownership(
  p_event_id       UUID,
  p_current_owner_id UUID,
  p_new_owner_id   UUID
) RETURNS JSONB LANGUAGE plpgsql AS $$
BEGIN
  -- Verify caller is OWNER
  IF NOT EXISTS (
    SELECT 1 FROM wedding.event_members
    WHERE event_id = p_event_id AND user_id = p_current_owner_id
      AND event_role = 'OWNER' AND invite_status = 'ACCEPTED'
  ) THEN
    RAISE EXCEPTION 'INSUFFICIENT_PERMISSION: Only OWNER can transfer ownership';
  END IF;

  -- New owner must be an accepted member
  IF NOT EXISTS (
    SELECT 1 FROM wedding.event_members
    WHERE event_id = p_event_id AND user_id = p_new_owner_id
      AND invite_status = 'ACCEPTED'
  ) THEN
    RAISE EXCEPTION 'USER_NOT_MEMBER: New owner must already be a member';
  END IF;

  -- Demote current owner to LEADER
  UPDATE wedding.event_members SET event_role = 'LEADER'
  WHERE event_id = p_event_id AND user_id = p_current_owner_id;

  -- Promote new owner
  UPDATE wedding.event_members SET event_role = 'OWNER'
  WHERE event_id = p_event_id AND user_id = p_new_owner_id;

  -- Update event created_by
  UPDATE wedding.wedding_events SET created_by = p_new_owner_id
  WHERE id = p_event_id;

  RETURN jsonb_build_object('ownership_transferred', TRUE);
END; $$;

## EVENT MIDDLEWARE (event.middleware.ts)

### requireEventRole(...roles):
- Extract eventId from req.params.eventId
- Extract userId from req.user (set by JWT middleware)
- Query wedding.event_members for this user+event combination
- Check event_role is in allowed roles
- Attach req.eventRole to request
- Throw 403 if insufficient role

```typescript
export const requireEventRole = (...allowedRoles: string[]) =>
  async (req, res, next) => {
    const { eventId } = req.params;
    const userId = req.user.userId;

    const result = await prisma.$queryRaw`
      SELECT event_role, invite_status
      FROM wedding.event_members
      WHERE event_id = ${eventId}::UUID
        AND user_id = ${userId}::UUID
    `;

    const member = result[0];

    if (!member || member.invite_status !== 'ACCEPTED') {
      return res.status(403).json({
        success: false,
        error: { code: 'NOT_A_MEMBER', message: 'You are not a member of this event' }
      });
    }

    if (!allowedRoles.includes(member.event_role)) {
      return res.status(403).json({
        success: false,
        error: { code: 'INSUFFICIENT_EVENT_ROLE', message: 'You do not have permission for this action' }
      });
    }

    req.eventRole = member.event_role;
    next();
  };
```

## EVENT ROLE GUARD COMPONENT

```typescript
// Hides children if user doesn't have required event role
const EventRoleGuard = ({ allowedRoles, children, fallback = null }) => {
  const { myEventRole } = useEventContext();
  const roleHierarchy = {
    OWNER: 5, LEADER: 4, EDITOR: 3, CONTRIBUTOR: 2, VIEWER: 1
  };
  const hasAccess = allowedRoles.some(
    role => roleHierarchy[myEventRole] >= roleHierarchy[role]
  );
  return hasAccess ? children : fallback;
};
```
